CREATE DATABASE deliciacupcakes

// primeras tablas a crear:

CREATE TABLE contacto(
    id_contacto int AUTO_INCREMENT,
    nombre varchar(80),
    apellido varchar(80),
    genero varchar(20),
    gmail varchar(80),
    como_nos_conociste varchar(200),
    CONSTRAINT pk_ic PRIMARY KEY(id_contacto)
);

CREATE TABLE categorias (
    id_categoria int AUTO_INCREMENT,
    categoria varchar(80),
    CONSTRAINT pk_ic PRIMARY KEY(id_categoria)
);

CREATE TABLE sucursales (
    id_sucursal int AUTO_INCREMENT,
    direccion varchar(200),
    vigencia boolean,
    CONSTRAINT pk_is PRIMARY KEY(id_sucursal)
);


// tablas a crear ultimas por tener referencias

CREATE TABLE productos (
    id_productos int AUTO_INCREMENT,
    nombre varchar(80),
    precio int,
    stock int,
    id_categoria int,
    CONSTRAINT pk_ip PRIMARY KEY (id_productos),
    CONSTRAINT fk_ic FOREIGN KEY(id_categoria) REFERENCES categorias(id_categoria)
);
